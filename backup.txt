    function updateScrollSpeed(val) { scrollSpeed = parseFloat(val); }

    function autoScrollLoop(timestamp) {
        if (!lastScrollTime) lastScrollTime = timestamp;
        const delta = timestamp - lastScrollTime;
        lastScrollTime = timestamp;
        scrollAccumulator += (20 * scrollSpeed) * (delta / 1000);
        if (scrollAccumulator >= 1) {
            const step = Math.floor(scrollAccumulator);
            document.getElementById('canvasWrap').scrollBy({ top: step });
            scrollAccumulator -= step;
        }
        autoScrollReq = requestAnimationFrame(autoScrollLoop);
    }

    function toggleAutoScroll() {
        const btnS = document.getElementById('btnScroll');
        const iconPlay = document.getElementById('iconPlay'), iconPause = document.getElementById('iconPause');
        const speedCtrl = document.getElementById('speedCtrl');
        if (autoScrollReq) {
            cancelAnimationFrame(autoScrollReq); autoScrollReq = null;
            if(btnS) btnS.classList.remove('gold-on');
            if (iconPlay) { iconPlay.style.display = 'block'; iconPause.style.display = 'none'; }
            if(speedCtrl) speedCtrl.classList.remove('active');
        } else {
            if(btnS) btnS.classList.add('gold-on');
            if (iconPlay) { iconPlay.style.display = 'none'; iconPause.style.display = 'block'; }
            if(speedCtrl) speedCtrl.classList.add('active');
            scrollSpeed = parseFloat(document.getElementById('speedSlider').value);
            lastScrollTime = performance.now();
            autoScrollReq = requestAnimationFrame(autoScrollLoop);
        }
    }

    function setupPageTracking() {
        const cw = document.getElementById('canvasWrap');
        cw.addEventListener('scroll', () => {
            const pages = document.getElementById('pdfContainer').querySelectorAll('.pdf-page-wrap');
            const cwRect = cw.getBoundingClientRect();
            let closest = 0, minDist = Infinity;
            pages.forEach((pg, i) => {
                const dist = Math.abs(pg.getBoundingClientRect().top - cwRect.top);
                if (dist < minDist) { minDist = dist; closest = i; }
            });
            currentPage = closest + 1; updatePageNav();
        }, { passive: true });
    }

            const res = await fetch(`${SCRIPT_URL}?action=download&id=${song.id}`);
            const data = await res.json();
            if (!data.data) throw new Error('Missing PDF data');
            const raw = atob(data.data);
            const bytes = new Uint8Array(raw.length);
            for (let i = 0; i < raw.length; i++) bytes[i] = raw.charCodeAt(i);
            pdfDoc = await pdfjsLib.getDocument({ data: bytes }).promise;
            totalPages = pdfDoc.numPages;
            container.innerHTML = '';
            container.classList.toggle('two-up', twoUpMode);
            const renderTasks = [];
            for (let p = 1; p <= totalPages; p++) { renderTasks.push(renderPage(p)); }
            await Promise.all(renderTasks);
            updatePageNav(); setTool('pan');
            try {
                const noteRes = await fetch(`${SCRIPT_URL}?action=getNotes&pdfId=${song.id}`);
                currentNotes = await noteRes.json();
                applyAnnotations();
            } catch (err) { console.log('No previous annotations found.'); }
        } catch (err) {
            container.innerHTML = `<div class="err-msg">Error loading document:<br>${err.message}</div>`;
        }
    }

    async function renderPage(pageNum) {
        const page = await pdfDoc.getPage(pageNum);
        const container = document.getElementById('pdfContainer');
        let targetWidth = twoUpMode ? Math.floor((container.clientWidth - 24) / 2) : Math.min(container.clientWidth - 4, 900);
        const baseVp = page.getViewport({ scale: 1 });
        const scale = targetWidth / baseVp.width;
        const vp = page.getViewport({ scale });
        const wrap = document.createElement('div');
        wrap.className = 'pdf-page-wrap fade-up';
        wrap.style.animationDelay = ((pageNum - 1) * 0.05) + 's';
        wrap.style.width = vp.width + 'px'; wrap.style.height = vp.height + 'px';
        wrap.dataset.page = pageNum;
        let inserted = false;
        for (let sibling of container.children) {
            if (parseInt(sibling.dataset.page) > pageNum) {
                container.insertBefore(wrap, sibling);
                inserted = true;
                break;
            }
        }
        if (!inserted) container.appendChild(wrap);
        const dpr = Math.min(window.devicePixelRatio || 1, 2);
        const pdfCvs = document.createElement('canvas');
        pdfCvs.style.width = vp.width + 'px'; pdfCvs.style.height = vp.height + 'px';
        pdfCvs.width = vp.width * dpr; pdfCvs.height = vp.height * dpr;
        const pdfCtx = pdfCvs.getContext('2d'); pdfCtx.scale(dpr, dpr);
        wrap.appendChild(pdfCvs);
        const drawCvs = document.createElement('canvas');
        drawCvs.className = 'draw-layer';
        drawCvs.style.width = vp.width + 'px'; drawCvs.style.height = vp.height + 'px';
        drawCvs.width = vp.width; drawCvs.height = vp.height;
        wrap.appendChild(drawCvs);
        const ctx = drawCvs.getContext('2d'); drawCtxs[pageNum] = ctx;
        attachDraw(drawCvs, ctx);
        await page.render({ canvasContext: pdfCtx, viewport: vp }).promise;
    }

    function attachDraw(canvas, ctx) {
        let drawing = false;
        const pos = (e) => {
            const r = canvas.getBoundingClientRect();
            const cx = e.touches ? e.touches[0].clientX : e.clientX;
            const cy = e.touches ? e.touches[0].clientY : e.clientY;
            return { x: (cx - r.left) * (canvas.width / r.width), y: (cy - r.top) * (canvas.height / r.height) };
        };
        const down = (e) => {
            if (currentTool === 'pan') return;
            drawing = true; const { x, y } = pos(e);
            ctx.beginPath(); ctx.moveTo(x, y); if (e.cancelable) e.preventDefault();
        };
        const move = (e) => {
            if (!drawing || currentTool === 'pan') return;
            const { x, y } = pos(e);
            ctx.lineCap = 'round'; ctx.lineJoin = 'round';
            if (currentTool === 'eraser') { ctx.globalCompositeOperation = 'destination-out'; ctx.lineWidth = currentThickness * 10; ctx.strokeStyle = 'rgba(0,0,0,1)'; }
            else if (currentTool === 'highlight') { ctx.globalCompositeOperation = 'source-over'; ctx.lineWidth = currentThickness * 6; ctx.strokeStyle = currentColor + '66'; }
            else { ctx.globalCompositeOperation = 'source-over'; ctx.lineWidth = currentThickness; ctx.strokeStyle = currentColor; }
            ctx.lineTo(x, y); ctx.stroke(); ctx.beginPath(); ctx.moveTo(x, y);
            if (e.cancelable) e.preventDefault();
        };
        const up = () => { drawing = false; ctx.globalCompositeOperation = 'source-over'; };
        canvas.addEventListener('mousedown', down); canvas.addEventListener('mousemove', move); canvas.addEventListener('mouseup', up); canvas.addEventListener('mouseleave', up);
        canvas.addEventListener('touchstart', down, { passive: false }); canvas.addEventListener('touchmove', move, { passive: false }); canvas.addEventListener('touchend', up);
    }

    async function executeUpload() {
        const fileInput = document.getElementById('uploadFileInput');
        const folderName = document.getElementById('uploadFolderInput').value.trim() || 'General';
        if(fileInput.files.length === 0) return alert('Please select a file to upload.');
        const file = fileInput.files[0];
        const btn = document.getElementById('uploadActionBtn');
        const origText = btn.innerText;
        btn.innerHTML = '<span class="spinner"></span> Uploading...'; btn.disabled = true;
        const reader = new FileReader();
        reader.onload = async function() {
            try {
                const req = await fetch(SCRIPT_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({ action: 'uploadFile', destFolder: folderName, fileName: file.name, data: reader.result })
                });
                const res = await req.json();
                if(res.success) {
                    closeModal('uploadModal'); await loadLibrary();
                    const uploadedSong = allSongs.find(s => s.id === res.id);
                    if(uploadedSong) loadSong(null, uploadedSong, uploadedSong.name.replace(/\.[^/.]+$/, ''), res.category);
                } else throw new Error(res.error);
            } catch (err) { alert('Upload failed: ' + err.message); } finally { btn.innerHTML = origText; btn.disabled = false; }
        };
        reader.readAsDataURL(file);
    }

    async function executeMove() {
        if(!currentFile) return;
        const newFolder = document.getElementById('moveFolderInput').value.trim() || 'General';
        const btn = document.getElementById('moveActionBtn');
        const origText = btn.innerText;
        btn.innerHTML = '<span class="spinner"></span> Moving...'; btn.disabled = true;
        try {
            const req = await fetch(SCRIPT_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                body: JSON.stringify({ action: 'moveFile', fileId: currentFile.id, destFolder: newFolder })
            });
            const res = await req.json();
            if(res.success) {
                currentFile.category = res.category;
                document.getElementById('songFolder').textContent = res.category;
                closeModal('moveModal'); await loadLibrary();
            } else throw new Error(res.error);
        } catch (err) { alert('Move failed: ' + err.message); } finally { btn.innerHTML = origText; btn.disabled = false; }
    }

    async function saveAnnotations() {
        if (!currentFile || !pdfDoc) return;
        const btnList = document.querySelectorAll('.save-btn');
        const origHTMLs = Array.from(btnList).map(b => b.innerHTML);
        btnList.forEach(btn => { btn.innerHTML = '<span class="spinner" style="border-top-color:var(--gold)"></span>'; btn.style.pointerEvents = 'none'; });
        try {
            for (let pg in drawCtxs) { currentNotes[pg] = drawCtxs[pg].canvas.toDataURL('image/png'); }
            const saveReq = await fetch(SCRIPT_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'text/plain;charset=utf-8' }, 
                body: JSON.stringify({ action: 'saveNotes', pdfId: currentFile.id, notes: currentNotes })
            });
            const saveRes = await saveReq.json();
            if (saveRes.success) {
                btnList.forEach(btn => { btn.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" style="width:16px;height:16px"><polyline points="20 6 9 17 4 12"/></svg>'; });
                setTimeout(() => { btnList.forEach((btn, i) => { btn.innerHTML = origHTMLs[i]; btn.style.pointerEvents = 'auto'; }); }, 2000);
            } else { throw new Error(saveRes.error); }
        } catch (e) {
            alert('Save failed: ' + e.message);
            btnList.forEach((btn, i) => { btn.innerHTML = origHTMLs[i]; btn.style.pointerEvents = 'auto'; });
        }
    }
